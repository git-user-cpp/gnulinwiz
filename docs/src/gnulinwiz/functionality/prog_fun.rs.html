65/// Validates that the program is not running with root privileges.
66///
67/// Checks the user ID to ensure the program is not executed as root, as running
68/// with root privileges is considered unsafe and unnecessary. If root privileges
69/// are detected, an error message is printed, and the program exits with status code 1.
70///
71/// # Panics
72/// Exits the program with status code 1 if the user ID is 0 (root).
73pub fn validate_root_priviliges() {
74    if unsafe { libc::getuid() } == 0 {
75        eprintln!("{}", "this program is not recommended to run with root privileges. please run it with your current user.\nexample: ./gnulinwiz".red());
76        // exits if root
77        exit(1);
78    }
79}

103/// Prompts the user to choose between default or custom software installation lists.
104///
105/// Asks the user to input a number: 0 for a custom list or any other number for the default list.
106/// Returns `true` for a custom list and `false` for the default list. Invalid input triggers
107/// an error and program exit via `handle_error`.
108///
109/// # Returns
110/// * `true` if the user selects a custom list (input is 0).
111/// * `false` if the user selects the default list (input is any non-zero number).
112///
113/// # Panics
114/// Exits the program via `handle_error` if the input cannot be parsed as an integer.
115///
116/// # Examples
117/// ```
118/// // Simulating user input of "0" would return true
119/// // Simulating user input of "1" would return false
120/// let use_custom = check_sw_install_type();
121/// ```
122pub fn check_sw_install_type() -> bool {
123    println!(
124        "{}",
125        "enter any number for the default list of software or 0 to enter a custom list:".yellow()
126    );
127
128    let input = read_input();
129
130    match input.trim().parse::<i8>() {
131        Ok(value) if value == 0 => {
132            println!("{}", "you chose to enter a custom list.\ninstallation takes a few minutes, please wait...".green());
133            return true;
134        }
135        Ok(_) => {
136            println!("{}", "you chose to use the default list.\ninstallation takes a few minutes, please wait...".green());
137            return false;
138        }
139        Err(_) => handle_error(
140            "please enter 0 for a custom list or any other number for the default list.",
141        ),
142    }
143}

145// Reads a line of input from stdin and returns it as a String.
146// Handles read errors by calling handle_error and exiting the program.
147fn read_input() -> String {
148    let mut input = String::new();
149    if let Err(error) = stdin().read_line(&mut input) {
150        handle_error(&format!("{}{}", "error reading input: ".red(), error))
151    }
152    return input;
153}
154
155/// Handles errors by printing a message and exiting the program.
156///
157/// Prints the provided error message in red, followed by a call to `print_setup_status_failed`,
158/// and terminates the program with exit code 1.
159///
160/// # Arguments
161/// * `message` - The error message to display.
162///
163/// # Panics
164/// Always exits the program with status code 1 after printing the error message.
165pub fn handle_error(message: &str) -> ! {
166    eprintln!("{}{}", "error: ".red(), message);
167    print_setup_status_failed();
168    exit(1);
169}

171/// Collects a list of software packages from user input for installation.
172///
173/// Prompts the user to enter a space-separated list of software package names,
174/// reads the input, and splits it into a vector of strings. Input errors are handled
175/// via the `read_input` function.
176///
177/// # Returns
178/// A vector of owned `String`s containing the package names entered by the user.
179///
180/// # Examples
181/// ```
182/// // Simulating user input of "vim nano" would return vec!["vim".to_string(), "nano".to_string()]
183/// let packages = set_sw_list();
184/// ```
185pub fn set_sw_list() -> Vec<String> {
186    println!("enter the software packages to install (separated by spaces):");
187    let input = read_input();
188    return input.trim().split_whitespace().map(String::from).collect();
189}
